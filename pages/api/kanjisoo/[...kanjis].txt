import { NextApiRequest, NextApiResponse } from "next";

export default async function KanjiHandler(
	req: NextApiRequest,
	res: NextApiResponse
) {
	const kanjis: string[] = [...req.query.kanjis];
	const endpoint = kanjis.join("/");
	try {
		switch (endpoint) {
			case "/kanjis":
				res.status(200).json({
					success: true,
					method: req.method,
					endpoint,
				});
				break;
			case "/kanjis/add":
				res.status(200).json({
					success: true,
					method: req.method,
					endpoint,
				});
				break;

			default:
				break;
		}
	} catch (error) {
		console.log(error);
	}
}
